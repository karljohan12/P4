const int PIN1 = 1;
const int PIN2 = 2;


function void setup() BEGIN
Servo base = PIN1;
Servo arm = PIN2;

Robot throwingArm = {base, arm};

ServoPosition waitingPosition = {0, 0};
ServoPosition throwAtWall = {10, 120};
ServoPosition targetTable = {base: 70};
ServoPosition makeThrow = {arm: 120}

END

function void loop() BEGIN

throwingArm(waitingPosition);
delay(1000);
throwingArm(throwAtWall);
delay(1000);

throwingArm(waitingPosition);
delay(1000)
throwingArm(targetTable);
delay(1000);
throwingArm(makeThrow);
delay(1000);

END
