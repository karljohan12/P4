int i = 0;

function void setup() BEGIN



Servo base = 11;
Servo shoulder = 10+i;
Servo elbow = 9;
Servo wristver = 6;
Servo wristrot = 5;
Servo gripper = 3;

Robot Braccio = {base, shoulder, elbow, wristver, wristrot, gripper};
ServoPosition standUpright = {10, 90, 100, 100, 100, 60};
ServoPosition Extend = {10, 90, 100, 100, 100, 30};

END

function void loop() BEGIN
delay(1000);
Braccio(standUpright);
delay(1000);
Braccio(Extend);

END
